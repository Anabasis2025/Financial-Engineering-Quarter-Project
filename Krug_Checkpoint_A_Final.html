def calculate_momentum(data, date_idx, lookback):
    """Calculate momentum using only PRIOR data (no look-ahead)"""
    if date_idx <= lookback:
        return {ticker: -999 for ticker in data.columns}
    
    current_date = data.index[date_idx - 1] 
    past_date = data.index[date_idx - 1 - lookback]
    
    momentum = {}
    for ticker in data.columns:
        try:
            momentum[ticker] = (data.loc[current_date, ticker] / data.loc[past_date, ticker]) - 1
        except:
            momentum[ticker] = -999
    return momentum

def calculate_ma(data, period=200):
    """Calculate moving average"""
    return data.rolling(window=period).mean()

def select_top_per_sleeve(momentum, sleeves, top_n):
    """Select top N stocks per sleeve"""
    selected = []
    for sleeve_name, sleeve_stocks in sleeves.items():
        sleeve_mom = {s: momentum.get(s, -999) for s in sleeve_stocks if s in data.columns}
        top = sorted(sleeve_mom.items(), key=lambda x: x[1], reverse=True)[:top_n]
        selected.extend([s[0] for s in top])
    return selected

def volatility_parity_weights(returns, selected_stocks):
    """Calculate volatility parity weights"""
    if len(selected_stocks) == 0:
        return {}
    vols = returns[selected_stocks].std()
    vols = vols.replace(0, 1e-6) 
    inv_vols = 1 / vols
    weights = inv_vols / inv_vols.sum()
    return dict(zip(selected_stocks, weights))

def mean_variance_weights(returns, selected_stocks):
    """Calculate mean-variance optimal weights"""
    if len(selected_stocks) == 0:
        return {}
    
    ret = returns[selected_stocks]
    mu = ret.mean() * 252
    cov = ret.cov() * 252
    
    n = len(selected_stocks)
    init_guess = np.array([1/n] * n)
    bounds = tuple((0, 1) for _ in range(n))
    constraints = {'type': 'eq', 'fun': lambda x: np.sum(x) - 1}
    
    def neg_sharpe(weights):
        port_ret = np.dot(weights, mu)
        port_vol = np.sqrt(np.dot(weights, np.dot(cov, weights)))
        return -port_ret / port_vol if port_vol > 0 else 999
    
    try:
        result = minimize(neg_sharpe, init_guess, method='SLSQP', 
                         bounds=bounds, constraints=constraints)
        if result.success:
            return dict(zip(selected_stocks, result.x))
    except:
        pass
    
    return {s: 1/len(selected_stocks) for s in selected_stocks}

def calculate_portfolio_value(holdings, data, date):
    """Calculate portfolio value"""
    if not holdings:
        return 0
    return sum(holdings.get(t, 0) * data.loc[date, t] for t in holdings)

def calculate_metrics(portfolio_series, initial_capital):
    """Calculate performance metrics"""
    if len(portfolio_series) == 0:
        return {'total_return': 0, 'annual_return': 0, 'sharpe': 0, 
                'sortino': 0, 'max_dd': 0, 'volatility': 0}
    
    total_ret = (portfolio_series.iloc[-1] / initial_capital - 1) * 100
    days = len(portfolio_series)
    annual_ret = ((portfolio_series.iloc[-1] / initial_capital) ** (252 / days) - 1) * 100
    
    daily_ret = portfolio_series.pct_change().dropna()
    sharpe = (daily_ret.mean() / daily_ret.std()) * np.sqrt(252) if daily_ret.std() > 0 else 0
    
    downside_ret = daily_ret[daily_ret < 0]
    sortino = (daily_ret.mean() / downside_ret.std()) * np.sqrt(252) if len(downside_ret) > 0 and downside_ret.std() > 0 else 0
    
    cum_max = portfolio_series.expanding().max()
    dd = (portfolio_series - cum_max) / cum_max
    max_dd = dd.min() * 100
    
    vol = daily_ret.std() * np.sqrt(252) * 100
    
    return {
        'total_return': total_ret,
        'annual_return': annual_ret,
        'sharpe': sharpe,
        'sortino': sortino,
        'max_dd': max_dd,
        'volatility': vol
    }
